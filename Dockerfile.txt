# Use an older Python version fully compatible with aiohttp
FROM python:3.11-slim

# Set working directory
WORKDIR /app

# Copy project files into the container
COPY . .

# Install dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Expose a port (optional, if your proxy uses one)
EXPOSE 8080

# Command to start your proxy server
CMD ["python", "main.py"]
